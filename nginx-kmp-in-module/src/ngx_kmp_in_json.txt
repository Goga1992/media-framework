    max %uL
    avg %uL obj->sum / obj->count

    no_media_info %ui
    no_key %ui

    connection %uA obj->connection->number
    remote_addr %jV

    received_bytes %O
    skipped_frames %func-ngx_kmp_in_stats_skip_json &obj->skipped
    latency %func-ngx_kmp_in_stats_latency_json &obj->latency
